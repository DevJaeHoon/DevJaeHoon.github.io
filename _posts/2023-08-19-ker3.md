## 구조체 콜백함수와 연결
```c
#include <stdio.h>

void aaa()
{

}

void bbb()
{

}

void ccc()
{

}

struct MBC
{
	void(*one)();
	void(*two)();
	void(*three)();
};

struct TV {
    struct MBC go;
    struct MBC shoot;
};

int main()
{
	struct MBC mbc =
	{
		.one = aaa,
		.two = bbb,
		.three = ccc,
	};
    
    struct TV tv = {
        .go = {
            .one = aaa,
            .two = bbb,
        },
        .shoot = {
            .three = aaa,
        }
    };
    
	return 0;
}
```
